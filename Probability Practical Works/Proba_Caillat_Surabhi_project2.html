def AR(n):
    Yt = []
    T = []
    for j in range(n):
        Y = rnd.standard_cauchy(n)
        U = rnd.uniform(0,1,n)
        t = min_i(U,Y)
        T += [t]
        Yt += [Y[t]]
    return (T,Yt)
A = AR(100)

def AR_cercle(n):
