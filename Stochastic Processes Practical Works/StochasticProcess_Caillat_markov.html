def genmarkov(x, n, p):
    dim = len(p[0])
    y = np.arange(n)
    y[0] = x
    for i in range(1,n):
        y[i] = rnd.choice(a = dim, p = p[y[i-1]])
    return(y[-1])

#genmarkov(1, n, pla)

def vgenmarkov(x,n,p):
    dim = len(p[0])
    y = np.arange(n)
    y[0] = x
    for i in range(1,n):
        y[i] = rnd.choice(a = dim, p = p[y[i-1]])
    return(y)

def puis(x,n,p,y):
    N = len(p)
    A = np.zeros(N)
    A[x] = 1
    return np.dot(np.transpose(A), np.linalg.matrix_power(p,n))[y]

# matrice stochastique p
def probinv(p):
    w, v = np.linalg.eig(p.T)
    v1 = v[:,np.isclose(w, 1)]
    v1 = v1[:,0]
    PI = v1 / v1.sum()
    PI = PI.real
    return(PI)

pi_pla = probinv(pla)
print("La probabilit invariante pour pla est : ", pi_pla)
print(np.isclose(np.dot(pi_pla,pla), pi_pla))

pi_pt = probinv(pt)
print("La probabilit invariante pour pla est : ", pi_pt)
print(np.isclose(np.dot(pi_pt,pt), pi_pt))

Rappelez les conditions d'application de ce thorme. Indiquez pour chacun des exercices $4.7,4.12$ si vous avez le droit de l'appliquer et faites le, ventuellement, pour rpondre aux questions de ces exercices.

Vous devez dterminer la/les probabilits invariantes, la fonction $f$ dont vous prenez la moyenne temporelle (c'est un vecteur de la bonne dimension) et calculer la moyenne $<f,\pi>$.

Thorme ergodique : Soit X une chaine irrductible positive rcurrente. Alors, pour toute fonction f positive ou borne, presque srement : $\frac{1}{n} \sum_{k=1}^{n} f\left(X_{k}\right) \rightarrow \pi(f)$

Dans l'exercice 4.7, la chane de Markov est irrductible car tous les tats communiquent. Tous les tats sont rcurrents positifs car l'esprance du temps de premier retour en un tat duquel on est parti est finie. On peut donc appliquer le thorme ergodique en l'appliquant la fonction positive : $f(X_{k}) = 1_{(X_{k} \ge 2)} = 1 - 1_{(X_{k} = 1)}$ Numriquement, on trouve $\pi(f) = 0.625$
